CREATE TABLE ensayo.ctc_categoria
(
    id bigint NOT NULL DEFAULT nextval('ensayo.ctc_categoria_id_seq'::regclass),
    nombre_categoria character varying(255) COLLATE pg_catalog."default",
    descripcion character varying(255) COLLATE pg_catalog."default",
    id_diccionarios bigint,
    CONSTRAINT categoria_pkey PRIMARY KEY (id)
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

    OWNER to postgres;
